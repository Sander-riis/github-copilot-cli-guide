   - URL: `https://cdn.jsdelivr.net/npm/highlight.js@11.11.1/lib/core.min.js`

2. **bash language pack** — exports a function, capture as variable:
   - URL: `https://cdn.jsdelivr.net/npm/highlight.js@11.11.1/lib/languages/bash.min.js`
   - The min.js files use UMD format. After pasting, the export registers as a module. For inline use, wrap each language file in an IIFE and capture the result:
     ```js
     const hljsBash = (() => { /* pasted bash.min.js content here */ ; return module.exports; })();
     ```
   - ALTERNATIVE simpler approach: The jsdelivr CDN also has ESM builds. However, the simplest reliable approach for inline use is:
     - Fetch the language file
     - Look at how it exports (it exports a function). For `core.min.js` + language files from v11.11.1, the language files export their language definition function via `module.exports`
     - Wrap in: `var hljsBash = function(hljs) { /* language function body extracted from bash.min.js */ }`
   - **EASIEST approach (recommended)**: Use the CDN paths that return the function directly. The jsdelivr `lib/languages/bash.min.js` contains something like:
     ```js
     var e={};e.exports=function(e){ /* language definition */ };
     ```
     Wrap the whole file content in:
     ```js
     const hljsBash = (function() { var module={exports:{}}; /* pasted content */; return module.exports; })();
     ```

## Part B: Application JS

After all the inlined library code, write this application JS verbatim:

```javascript
/* ── LANGUAGE REGISTRATION ── */
hljs.registerLanguage('bash',     hljsBash);
hljs.registerLanguage('yaml',     hljsYaml);
hljs.registerLanguage('json',     hljsJson);
hljs.registerLanguage('markdown', hljsMarkdown);
hljs.registerLanguage('shell',    hljsShell);

/* ── TAB NAVIGATION ── */
function initTabs() {
  const tabNav  = document.querySelector('.tab-nav');
  const panels  = document.querySelectorAll('.content-panel');
  const sidebars = document.querySelectorAll('.sidebar-nav');

  tabNav.addEventListener('click', (e) => {
    const tab = e.target.closest('[data-tab]');
    if (!tab) return;
    const target = tab.dataset.tab;

    // Update ARIA states
    tabNav.querySelectorAll('[role="tab"]').forEach(t => {
      const active = t.dataset.tab === target;
      t.classList.toggle('tab-nav__tab--active', active);
      t.setAttribute('aria-selected', String(active));
    });

    // Show/hide panels — use requestAnimationFrame to ensure display:block
    // is computed BEFORE the opacity transition class is applied.
    // Without this, display:none → display:block + opacity change happens
    // in the same frame and the transition is skipped (Pitfall 4 from RESEARCH.md).
    panels.forEach(panel => {
      const isActive = panel.id === `section-${target}`;
      if (isActive) {
        panel.classList.remove('content-panel--hidden');
        requestAnimationFrame(() => panel.classList.add('content-panel--active'));
      } else {
        panel.classList.remove('content-panel--active');
        panel.classList.add('content-panel--hidden');
      }
    });

    // Show correct sidebar nav
    sidebars.forEach(nav => {
      nav.classList.toggle('sidebar-nav--hidden', nav.id !== `sidebar-nav-${target}`);
    });

    // Reset scroll position on tab switch
    document.querySelector('.main-content').scrollTop = 0;
    document.querySelector('.sidebar').scrollTop = 0;
  });
}

/* ── COPY TO CLIPBOARD ── */
function initCopyButtons() {
  // Event delegation — catches buttons in all panels including hidden ones
  document.addEventListener('click', async (e) => {
    const btn = e.target.closest('.code-block__copy');
    if (!btn) return;

    // CRITICAL: Use .textContent NOT .innerText.
    // .innerText is layout-aware and returns empty string for elements
    // inside display:none containers (hidden tab panels). Pitfall 5 in RESEARCH.md.
    const code = btn.closest('.code-block').querySelector('code').textContent;

    try {
      await navigator.clipboard.writeText(code);
    } catch {
      // Fallback for environments where clipboard API is blocked
      const ta = document.createElement('textarea');
      ta.value = code;
      ta.style.cssText = 'position:fixed;top:-9999px;opacity:0;';
      document.body.appendChild(ta);
      ta.select();
      document.execCommand('copy');
      document.body.removeChild(ta);
    }

    // Visual success state — 2000ms per UI-SPEC.md
    const label    = btn.querySelector('.code-block__copy-label');
    const iconCopy = btn.querySelector('.icon--copy');
    const iconCheck = btn.querySelector('.icon--check');

    btn.classList.add('code-block__copy--copied');
    if (label)     label.textContent = 'Copied!';
    if (iconCopy)  iconCopy.classList.add('icon--hidden');
    if (iconCheck) iconCheck.classList.remove('icon--hidden');

    setTimeout(() => {
      btn.classList.remove('code-block__copy--copied');
      if (label)     label.textContent = 'Copy code';
      if (iconCopy)  iconCopy.classList.remove('icon--hidden');
      if (iconCheck) iconCheck.classList.add('icon--hidden');
    }, 2000);
  });
}

/* ── SCROLLSPY ── */
function initScrollspy() {
  // rootMargin: '-56px 0px 0px 0px' accounts for the 56px sticky header.
  // Without this offset, sections trigger as visible before they scroll
  // into the visible viewport area below the header (Pitfall 6 in RESEARCH.md).
  const observer = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
      const id   = entry.target.id;
      const link = document.querySelector(`.sidebar-nav__link[href="#${id}"]`);
      if (!link) return;

      if (entry.isIntersecting) {
        // Deactivate all links in currently visible sidebar
        const activeSidebar = document.querySelector('.sidebar-nav:not(.sidebar-nav--hidden)');
        activeSidebar?.querySelectorAll('.sidebar-nav__link--active')
          .forEach(l => l.classList.remove('sidebar-nav__link--active'));

        link.classList.add('sidebar-nav__link--active');
      }
    });
  }, {
    threshold:  0.4,
    rootMargin: '-56px 0px 0px 0px'
  });

  // Observe all sections in all panels (including hidden ones — observer handles visibility)
  document.querySelectorAll('.content-section[id]').forEach(s => observer.observe(s));
}

/* ── INIT ── */
document.addEventListener('DOMContentLoaded', () => {
  hljs.highlightAll();
  initTabs();
  initCopyButtons();
  initScrollspy();
});
```
