  $status = if ($found) { "PASS" } else { "FAIL" }
  if (-not $found) { $allPass = $false }
  Write-Host "$($check.Key): $status"

  Write-Host "FAIL: Old em-dash headings at lines: $($oldHeadings.LineNumber -join ', ')"

} else {
  Write-Host "Heading normalization: PASS"
}

# 6. Security warning strength check
$strongWarning = Select-String -Path "index.html" -Pattern "run any command on your machine" -Quiet
Write-Host "Security warning strength: $(if ($strongWarning) { 'PASS — strong language confirmed' } else { 'WARN — verify warning text manually' })"

$priorChecks = [ordered]@{
  "CLI tab intact"    = "standalone terminal AI agent"

  $status = if ($found) { "PASS" } else { "FAIL — REGRESSION" }
  if (-not $found) { $allPass = $false }
  Write-Host "$($check.Key): $status"
}

if ($allPass -and $dateCheck) {
  Write-Host "`nAll automated checks PASSED. Proceeding to human verification."
} else {
  Write-Host "`nSome checks FAILED. Fix issues before human verification."
}
